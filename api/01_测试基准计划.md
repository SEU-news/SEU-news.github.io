            'realname': '测试用户',

    def test_login(self):
        """测试登录"""
        response = requests.post(f'{BASE_URL}/auth/login/', json={
            'username': 'test_user',
            'password': 'test123'
        })
        assert response.status_code == 200
        data = response.json()
        assert data['success'] == True
        assert 'user' in data

    def test_current_user(self):
        """测试获取当前用户"""
